## Validación

<CodeRunner language="python" title="Validación de formularios" initialCode={`def validar_formulario(datos, reglas):
    """Valida datos contra un diccionario de reglas (funciones lambda)."""
    errores = []
    for campo, regla in reglas.items():
        valor = datos.get(campo, "")
        if not regla(valor):
            errores.append(campo)
    return errores

# Definir reglas con lambdas
reglas = {
    "nombre": lambda v: isinstance(v, str) and len(v) >= 2,
    "email": lambda v: isinstance(v, str) and "@" in v and "." in v,
    "edad": lambda v: isinstance(v, int) and 0 < v < 120,
    "password": lambda v: isinstance(v, str) and len(v) >= 8,
}

form_ok = {"nombre": "Ana", "email": "ana@mail.com", "edad": 28, "password": "MiClave123"}
form_mal = {"nombre": "A", "email": "sin-arroba", "edad": -5, "password": "abc"}

errores1 = validar_formulario(form_ok, reglas)
errores2 = validar_formulario(form_mal, reglas)

print(f"Form OK — Errores: {errores1 or 'ninguno'}")
print(f"Form mal — Errores: {errores2}")`} />

## Transformación de datos

<CodeRunner language="python" title="Pipeline de transformación" initialCode={`# Datos de ventas mensuales
ventas_raw = [
    ("2024-01", 15000), ("2024-02", 18000), ("2024-03", 12000),
    ("2024-04", 22000), ("2024-05", 19500), ("2024-06", 25000),
]

meses_nombre = {
    "01": "Enero", "02": "Febrero", "03": "Marzo",
    "04": "Abril", "05": "Mayo", "06": "Junio",
}

# Transformar a formato legible
transformadas = list(map(
    lambda v: {
        "mes": meses_nombre[v[0].split("-")[1]],
        "ventas": v[1],
        "formateado": f"{v[1]:,}€".replace(",", "."),
    },
    ventas_raw
))

# Agregar indicador de rendimiento
media = sum(v[1] for v in ventas_raw) / len(ventas_raw)
for v in transformadas:
    icono = "[ASCENDENTE]" if v["ventas"] >= media else "[DESCENDENTE]"
    print(f"  {icono} {v['mes']}: {v['formateado']}")

print(f"\\n  Media: {round(media):,}€".replace(",", "."))`} />
